port debug:
  channel DIO
  channel CLK
  channel SWO
  channel TX
  channel RX
  color "brown"

  config "SWD":
    DIO = *_SWDIO
    CLK = *_SWCLK
    SWO = *_SWO
    TX = USART*_TX
    RX = USART*_RX
    require same_instance(TX, RX, "USART")

port MOT:
  channel A
  channel AN
  channel B
  channel BN
  channel C
  channel CN
  channel BRAKE
  channel FAN
  channel STO0
  channel STO1
  channel IA
  channel IB
  channel IC
  channel IDC
  channel UA
  channel UB
  channel UC
  channel UDC
  channel IBRAKE
  channel TBRAKE
  channel TMOT_A
  channel TMOT_B
  channel TMOT_C
  channel TFET
  channel THEATSINK
  color "blue"

  config "MOT":
    # pwm
    A = TIM[1,8]_CH1
    AN = TIM[1,8]_CH1N
    B = TIM[1,8]_CH2
    BN = TIM[1,8]_CH2N
    C = TIM[1,8]_CH3
    CN = TIM[1,8]_CH3N
    BRAKE = TIM[1,8]_CH4
    FAN = TIM[2-5]_CH*
    
    # phase currents + voltages
    IA = ADC1_INP[1-3]
    IB = ADC2_INP[1-3]
    IC = ADC3_INP[1-3]
    UA = ADC1_INP[1-6]
    UB = ADC2_INP[1-6]
    UC = ADC3_INP[1-6]
    
    #dc current + voltage
    IDC = ADC1_INP[7-9]
    UDC = ADC2_INP[7-9]
    
    # brake current + temperature
    IBRAKE = ADC1_INP[7-9]
    TBRAKE = ADC2_INP[10-15]
    
    # motor + mosfet temperature
    TMOT_A = ADC1_INP[10-15]
    TMOT_B = ADC2_INP[10-15]
    TMOT_C = ADC3_INP[10-15]
    TFET = ADC3_INP[10-15]
    THEATSINK = ADC3_INP[10-15]
    
    STO0 = TIM[1,8]_BKIN | TIM[1,8]_BKIN2
    STO1 = TIM[1,8]_BKIN | TIM[1,8]_BKIN2
    require same_instance(A, AN, B, BN, C, CN, STO0, STO1, "TIM")
