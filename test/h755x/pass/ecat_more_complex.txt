port MOT:
  channel A
  channel AN
  channel B
  channel BN
  channel C
  channel CN
  channel BRAKE
  channel FAN
  channel STO0
  channel STO1
  channel IA
  channel IB
  channel IC
  channel IDC
  channel UA
  channel UB
  channel UC
  channel UDC
  channel IBRAKE
  channel TBRAKE
  channel TMOT_A
  channel TMOT_B
  channel TMOT_C
  channel TFET
  channel THEAT
  color "blue"

  config "MOT":
    # pwm
    A = TIM[1,8]_CH1
    AN = TIM[1,8]_CH1N
    B = TIM[1,8]_CH2
    BN = TIM[1,8]_CH2N
    C = TIM[1,8]_CH3
    CN = TIM[1,8]_CH3N
    BRAKE = TIM[1,8]_CH4
    FAN = TIM*_CH*
    
    # phase currents + voltages
    IA = ADC1_INP[0-3]
    IB = ADC2_INP[0-3]
    IC = ADC3_INP[0-3]
    UA = ADC1_INP[1-6]
    UB = ADC2_INP[1-6]
    UC = ADC3_INP[1-6]
    
    #dc current + voltage
    IDC = ADC1_INP[7-18]
    UDC = ADC2_INP[7-18]
    
    # brake current + temperature
    IBRAKE = ADC3_INP[7-18]
    TBRAKE = ADC3_INP[7-18]
    
    # motor + mosfet temperature
    TMOT_A = ADC1_INP[7-18]
    TMOT_B = ADC2_INP[7-18]
    TMOT_C = ADC3_INP[7-18]
    TFET = ADC1_INP[7-18]
    THEAT = ADC2_INP[7-18]
    
    STO0 = TIM[1,8]_BKIN | TIM[1,8]_BKIN2
    STO1 = TIM[1,8]_BKIN | TIM[1,8]_BKIN2
    require same_instance(A, AN, B, BN, C, CN, STO0, STO1, BRAKE, "TIM")
